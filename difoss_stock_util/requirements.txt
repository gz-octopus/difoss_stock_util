# ---------------------------------------------------------------------------
# 核心依赖（REPL / CLI 框架）
# ---------------------------------------------------------------------------
rich >= 14.2.0           # 终端美化输出、进度条、Table渲染
click >= 8.3.3            # CLI 命令框架
prompt_toolkit >= 3.0.52   # REPL 交互模式（历史记录、Tab补全、语法高亮）
click-repl >= 0.3.0        # click 与 prompt_toolkit 的桥接层（ClickCompleter）
colorama                   # Windows 控制台 ANSI 颜色支持

# ---------------------------------------------------------------------------
# 数据与数据库
# ---------------------------------------------------------------------------
SQLAlchemy >= 2.0.44      # ORM 数据库引擎
pandas                     # 数据分析与处理（K线、行情数据）
numpy                      # 数值计算基础库

# ---------------------------------------------------------------------------
# 配置与工具
# ---------------------------------------------------------------------------
pyyaml                     # YAML 配置文件解析
python-dotenv              # .env 环境变量文件加载

# ---------------------------------------------------------------------------
# A 股市场专用
# ---------------------------------------------------------------------------
chinese_calendar           # 中国节假日判断（交易日计算）

# ---------------------------------------------------------------------------
# 可选依赖 — 按需安装
# ---------------------------------------------------------------------------

# 通达信技术分析（需要先安装 ta-lib C 库，参考 https://github.com/TA-Lib/ta-lib-python）
ta-lib

# 数据差异比较（metric_data/slb.py 中使用）
dictdiffer

# 图表绘制（metric_data/slb.py 中使用）
matplotlib

# xtquant / miniQMT 量化交易接口（需从迅投官网获取）
xtquant

# 时区处理（xtquant_util.py 中使用）
pytz
